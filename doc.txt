import sqlite3

# Conectando ao banco de dados (será criado se não existir)
conn = sqlite3.connect('meu_banco_de_dados.db')

# Criando um cursor
cursor = conn.cursor()

# Criando a tabela Produto
cursor.execute('''
CREATE TABLE IF NOT EXISTS Produto (
    Codigo INTEGER PRIMARY KEY AUTOINCREMENT,
    Nome TEXT NOT NULL,
    Valor_Pago REAL NOT NULL,
    Valor_Venda REAL NOT NULL,
    Data_Entrada TEXT NOT NULL,
    Validade TEXT
)
''')

# Criando a tabela Estoque
cursor.execute('''
CREATE TABLE IF NOT EXISTS Estoque (
    Codigo_Produto_FK INTEGER NOT NULL,
    Quantidade_Entrada INTEGER NOT NULL,
    Quantidade_Saida INTEGER NOT NULL,
    Quantidade_Atual INTEGER NOT NULL,
    Data_Registro TEXT NOT NULL,
    FOREIGN KEY (Codigo_Produto_FK) REFERENCES Produto (Codigo)
)
''')

# Confirmando as operações
conn.commit()

# Fechando a conexão
conn.close()
